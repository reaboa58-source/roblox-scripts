نظام أوامر الأدمن (Admin Commands) صعب

سكربت متكامل لنظام أوامر الأدمن! يعطيك تحكم كامل بالسيرفر: طرد لاعبين، إعطاء أدوات، تغيير السرعة، تلفريب، وغيره.

🔵 ServerScriptService - Script 
-- نظام أوامر الأدمن المتكامل
local Players = game:GetService("Players")
local Commands = {}
local Admins = {"YourUsername"} -- ضع اسمك هنا

local function isAdmin(player)
    for _, name in ipairs(Admins) do
        if player.Name == name then return true end
    end
    return false
end

Commands.give = function(admin, args)
    local targetName = args[1]
    local toolName = args[2]
    local target = Players:FindFirstChild(targetName)
    if target and target.Character then
        local tool = game.ReplicatedStorage:FindFirstChild(toolName)
        if tool then
            tool:Clone().Parent = target.Backpack
            print("✅ تم إعطاء الأداة!")
        end
    end
end

Commands.kick = function(admin, args)
    local target = Players:FindFirstChild(args[1])
    if target then
        target:Kick("تم طردك من السيرفر")
    end
end

Commands.speed = function(admin, args)
    local target = Players:FindFirstChild(args[1])
    local speed = tonumber(args[2]) or 50
    if target and target.Character then
        target.Character.Humanoid.WalkSpeed = speed
    end
end

Commands.tp = function(admin, args)
    local target = Players:FindFirstChild(args[1])
    if target and target.Character and admin.Character then
        target.Character.HumanoidRootPart.CFrame = admin.Character.HumanoidRootPart.CFrame
    end
end

Players.PlayerAdded:Connect(function(player)
    player.Chatted:Connect(function(msg)
        if isAdmin(player) and msg:sub(1,1) == ";" then
            local args = msg:sub(2):split(" ")
            local cmd = table.remove(args, 1)
            if Commands[cmd] then
                Commands[cmd](player, args)
            end
        end
    end)
end)

📌 خطوات الاستخدام:

  1. افتح Roblox Studio وادخل على لعبتك
  2. روح لـ ServerScriptService وسوي Script جديد
  3. انسخ السكربت فوق والصقه
  4. غيّر "YourUsername" لاسمك في روبلوكس
  5. سوي RemoteEvent في ReplicatedStorage إذا تحتاج تواصل بين Client و Server
  6. في اللعبة اكتب في الشات: ;kick PlayerName أو ;speed PlayerName 100

نظام أنيميشنات متقدم متوسط

نظام أنيميشنات: مشي، جري، قفز، رقص، وإمotes مع نظام تشغيل سلس!

🔵 ServerScriptService - Script 
-- نظام أنيميشنات
local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

ReplicatedStorage
local AnimationsFolder = Instance.new("Folder")
AnimationsFolder.Name = "Animations"
AnimationsFolder.Parent = ReplicatedStorage



local AnimationSystem = {}

function AnimationSystem.SetupCharacter(player, character)
    local humanoid = character:WaitForChild("Humanoid")
    local animator = humanoid:FindFirstChildOfClass("Animator") or Instance.new("Animator", humanoid)

    local tracks = {}

    for _, anim in ipairs(AnimationsFolder:GetChildren()) do
        if anim:IsA("Animation") then
            tracks[anim.Name] = animator:LoadAnimation(anim)
        end
    end

    humanoid.Running:Connect(function(speed)
        if speed > 0.1 then
            if speed > 20 and tracks.Run then
                if tracks.Walk then tracks.Walk:Stop() end
                if not tracks.Run.IsPlaying then tracks.Run:Play() end
            elseif tracks.Walk then
                if tracks.Run then tracks.Run:Stop() end
                if not tracks.Walk.IsPlaying then tracks.Walk:Play() end
            end
        else
            if tracks.Walk then tracks.Walk:Stop() end
            if tracks.Run then tracks.Run:Stop() end
            if tracks.Idle and not tracks.Idle.IsPlaying then tracks.Idle:Play() end
        end
    end)

    humanoid.Jumping:Connect(function()
        if tracks.Jump then
            tracks.Jump:Play()
        end
    end)

    return tracks
end

Players.PlayerAdded:Connect(function(player)
    player.CharacterAdded:Connect(function(char)
        wait(1) -- انتظر تحميل الشخصية
        AnimationSystem.SetupCharacter(player, char)
    end)
end)

-- RemoteEvent للرقص والإمotes
local DanceEvent = Instance.new("RemoteEvent")
DanceEvent.Name = "DanceEvent"
DanceEvent.Parent = ReplicatedStorage

DanceEvent.OnServerEvent:Connect(function(player, danceName)
    local character = player.Character
    if character then
        local humanoid = character:FindFirstChild("Humanoid")
        local animator = humanoid and humanoid:FindFirstChildOfClass("Animator")
        if animator then
            local anim = AnimationsFolder:FindFirstChild(danceName)
            if anim then
                local track = animator:LoadAnimation(anim)
                track.Priority = Enum.AnimationPriority.Action
                track:Play()
            end
        end
    end
end)

    local character = tool.Parent
    local humanoid = character:FindFirstChild("Humanoid")

    if humanoid then
        local animator = humanoid:FindFirstChildOfClass("Animator")
        if animator then
            local attackAnim = Instance.new("Animation")
            attackAnim.AnimationId = "rbxassetid://YOUR_ANIMATION_ID"
            local track = animator:LoadAnimation(attackAnim)
            track:Play()
        end
    end

    local handle = tool:FindFirstChild("Handle")
    if handle then
        local hitbox = Instance.new("Part")
        hitbox.Size = Vector3.new(5, 5, 5)
        hitbox.CFrame = handle.CFrame * CFrame.new(0, 0, -3)
        hitbox.Anchored = true
        hitbox.CanCollide = false
        hitbox.Transparency = 1
        hitbox.Parent = workspace

        local connection
        connection = hitbox.Touched:Connect(function(hit)
            local targetHumanoid = hit.Parent:FindFirstChild("Humanoid")
            if targetHumanoid and targetHumanoid.Parent ~= character then
                targetHumanoid:TakeDamage(25)
                print(" ضربة ناجحة! الضرر: 25")

        end)

        game:GetService("Debris"):AddItem(hitbox, 0.3)
    end

    local sound = Instance.new("Sound")
    sound.SoundId = "rbxassetid://YOUR_SOUND_ID"
    sound.Parent = tool.Handle
    sound:Play()
end)

نظام حفظ البيانات (DataStore) صعب

سكربت احترافي لحفظ بيانات اللاعبين: فلوس، مستوى، عناصر، إحصائيات - مع نظام آمن ضد فقدان البيانات!

🔵 ServerScriptService - Script 
-- (DataStore)
local DataStoreService = game:GetService("DataStoreService")
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")

local PlayerData = DataStoreService:GetDataStore("PlayerData_v2")

local DEFAULT_DATA = {
    Coins = 0,
    Level = 1,
    XP = 0,
    Items = {},
    Stats = {
        Strength = 10,
        Speed = 10,
        Defense = 10
    },
    JoinDate = os.time()
}

local sessionData = {}
local dataLoaded = {}

local function safeSave(player, data)
    local success, err = pcall(function()
        PlayerData:SetAsync(player.UserId, data)
    end)
    if not success then
        warn(" فشل الحفظ لـ " .. player.Name .. ": " .. tostring(err))
        return false
    end
    print(" تم حفظ بيانات " .. player.Name)
    return true
end

local function safeLoad(player)
    local success, data = pcall(function()
        return PlayerData:GetAsync(player.UserId)
    end)

    if success and data then
        print(" تم تحميل بيانات " .. player.Name)
        return data
    else
        print(" بيانات جديدة لـ " .. player.Name)
        return table.clone(DEFAULT_DATA)
    end
end

Players.PlayerAdded:Connect(function(player)
    
    local data = safeLoad(player)
    sessionData[player.UserId] = data
    dataLoaded[player.UserId] = true

    local leaderstats = Instance.new("Folder")
    leaderstats.Name = "leaderstats"
    leaderstats.Parent = player

    local coins = Instance.new("IntValue")
    coins.Name = " Coins"
    coins.Value = data.Coins
    coins.Parent = leaderstats

    local level = Instance.new("IntValue")
    level.Name = "Level"
    level.Value = data.Level
    level.Parent = leaderstats

    player.CharacterAdded:Connect(function(char)
        wait(2)
        data.Coins = data.Coins + 10
        coins.Value = data.Coins
        print("💰 " .. player.Name .. " حصل على 10 فلوس!")
    end)
end)

Players.PlayerRemoving:Connect(function(player)
    if dataLoaded[player.UserId] then
        safeSave(player, sessionData[player.UserId])
        sessionData[player.UserId] = nil
        dataLoaded[player.UserId] = nil
    end
end)

while true do
    wait(60)
    for userId, data in pairs(sessionData) do
        local player = Players:GetPlayerByUserId(userId)
        if player then
            safeSave(player, data)
        end
    end
end

game:BindToClose(function()
    for _, player in ipairs(Players:GetPlayers()) do
        if dataLoaded[player.UserId] then
            safeSave(player, sessionData[player.UserId])
        end
    end
end)

لوحة المتصدرين (Leaderboard) سهل

لوحة متصدرين تبين أفضل اللاعبين في اللعبة مع تحديث فوري!

🔵 ServerScriptService - Script 
-- لوحة المتصدرين 
local Players = game:GetService("Players")

local playerStats = {}

Players.PlayerAdded:Connect(function(player)
    -- إنشاء leaderstats
    local leaderstats = Instance.new("Folder")
    leaderstats.Name = "leaderstats"
    leaderstats.Parent = player

    local kills = Instance.new("IntValue")
    kills.Name = "⚔️ Kills"
    kills.Value = 0
    kills.Parent = leaderstats

    local deaths = Instance.new("IntValue")
    deaths.Name = "💀 Deaths"
    deaths.Value = 0
    deaths.Parent = leaderstats

    local points = Instance.new("IntValue")
    points.Name = "🏆 Points"
    points.Value = 0
    points.Parent = leaderstats

    playerStats[player.UserId] = {
        Kills = kills,
        Deaths = deaths,
        Points = points
    }

    player.CharacterAdded:Connect(function(char)
        local humanoid = char:WaitForChild("Humanoid")
        humanoid.Died:Connect(function()
            deaths.Value = deaths.Value + 1

            local killer = humanoid:FindFirstChild("creator")
            if killer then
                local killerPlayer = killer.Value
                if killerPlayer and playerStats[killerPlayer.UserId] then
                    playerStats[killerPlayer.UserId].Kills.Value += 1
                    playerStats[killerPlayer.UserId].Points.Value += 100
                end
            end
        end)
    end)
end)

local function tagHumanoid(player, humanoid)
    local creator = Instance.new("ObjectValue")
    creator.Name = "creator"
    creator.Value = player
    creator.Parent = humanoid
end

📖 دليل استخدام السكربتات

الخطوة الأولى: فهم بنية روبلوكس ستوديو

قبل ما تستخدم أي سكربت، لازم تعرف الفرق بين الأماكن:

  • ServerScriptService - السكربتات اللي تشتغل على السيرفر (لوجيك اللعبة، حفظ البيانات)
  • StarterPlayerScripts - سكربتات تشتغل عند كل لاعب (كاميرا، حركة)
  • StarterGui - واجهات المستخدم (أزرار، قوائم)
  • ReplicatedStorage - مكان مشترك بين السيرفر واللاعب (أدوات، أنيميشنات)
  • Workspace - العالم اللي اللاعبين يشوفونه (أجزاء، شخصيات)

الخطوة الثانية: كيف تضيف سكربت

  1. افتح Roblox Studio وافتح لعبتك
  2. في Explorer (على اليمين)، دور على المكان المناسب
  3. اضغط Right Click → Insert Object → اختار Script أو LocalScript
  4. انسخ السكربت من الموقع والصقه في المحرر
  5. اضغط Play (F5) عشان تجرب!

للتواصل بين LocalScript و Script تحتاج RemoteEvents:

مثال على التواصل 
-- في ReplicatedStorage سوي RemoteEvent وسمّه "MyEvent"

local ReplicatedStorage = game:GetService("ReplicatedStorage")
local event = ReplicatedStorage:WaitForChild("MyEvent")

event:FireServer("Hello Server!", 100)

local ReplicatedStorage = game:GetService("ReplicatedStorage")
local event = ReplicatedStorage:WaitForChild("MyEvent")

event.OnServerEvent:Connect(function(player, message, number)
    print(player.Name .. " أرسل: " .. message)
    print("الرقم: " .. number)
end)

الخطوة الخامسة: إضافة أنيميشنات

  1. افتح Animation Editor من التبويب Avatar
  2. سوي أنيميشنك وارفعه (Publish to Roblox)
  3. خذ الـ Animation ID (الرقم اللي يطلع)
  4. سوي Animation object في ReplicatedStorage
  5. حط الـ ID في خاصية AnimationId
  6. استخدم السكربت اللي فوق عشن تشغلها!

نصيحة لاستخدام الذكاء الاصطناعي

لما تطلب سكربت من AI، خل السؤال مفصل جداً! اذكر:

  • وين تحط السكربت (ServerScriptService, StarterGui, إلخ)
  • إذا تحتاج DataStore ولا لا
  • GUI ولا بدون GUI
  • مستوى الصعوبة (سهل، متوسط، صعب)
  • أي سكربتات ثانية تشتغل معاه

نصائح بتفيدك

1. أمان اللعبة (Security)

  • Never trust the client! دايم تحقق من البيانات في السيرفر
  • استخدم pcall() مع DataStore عشان ما تخسر بيانات اللاعبين
  • Rate Limiting: حد عدد الطلبات (RemoteEvents) عشان ما يفجرون اللعبة
  • Sanitize inputs: تحقق من البيانات اللي يبعثها اللاعب
  • استخدم game.PrivateServerId عشان تعرف إذا اللعبة Private ولا لا

2. الأداء (Performance)

  • استخدم task.wait() بدل wait() - أسرع وأفضل
  • تجنب while true do الكثيرة، استخدم Events بدلها
  • Part count: حاول ما تزيد Parts في Workspace عن 10,000
  • استخدم StreamingEnabled للعوالم الكبيرة
  • أغلق الـ Connections اللي ما تحتاجها: connection:Disconnect()

3. التصميم (Design)

  • استخدم UIConstraint (UIAspectRatioConstraint) عشان الواجهة تتناسب مع كل شاشة
  • التباين: خل النص واضح على الخلفية (Contrast ratio 4.5:1 على الأقل)
  • Animations: استخدم TweenService بدل while loops للأنيميشنات
  • Consistency: خل الألوان والأزرار متناسقة في كل اللعبة
  • Feedback: دايم أعطِ اللاعب رد فعل لما يسوي شي (صوت، جسيمات)

4. التوافق (Compatibility)

  • اختبر لعبتك على Mobile و Console مو بس PC
  • استخدم UserInputService عشن تدعم كل الأجهزة
  • Mobile buttons: خل الأزرار كبيرة (min 44x44 pixels)
  • Text scaling: استخدم TextScaled = true

5. الاختبار (Testing)

  • اختبر مع 2+ لاعبين (Play → Play with up to 3 players)
  • اختبر DataStore بـ Studio API Access Enabled
  • اختبر edge cases: ماذا لو اللاعب خرج فجأة؟ ماذا لو السيرفر وقع؟
  • استخدم print() و warn() للـ Debugging
  • سجل الأخطاء في ملف خارجي لو تقدر
